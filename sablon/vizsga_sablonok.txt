================================

Ez egy általános, bemásolható jegyzet C# konzolos, C# WinForms és PHP API + webes feladatokhoz.
A kódok szándékosan egyszerűek, vizsgázós stílusban vannak írva.


1. C# OSZTÁLY SABLON - GETTER, SETTER, KONSTRUKTOR
==================================================

Példa: Sportolo.cs vagy Fuggohid.cs

public class Sportolo
{
    public int Helyezes { get; set; }
    public double Eredmeny { get; set; }
    public string Nev { get; set; }
    public string Orszagkod { get; set; }
    public string Helyszin { get; set; }
    public string Datum { get; set; }

    public Sportolo(int helyezes, double eredmeny, string nev, string orszagkod, string helyszin, string datum)
    {
        Helyezes = helyezes;
        Eredmeny = eredmeny;
        Nev = nev;
        Orszagkod = orszagkod;
        Helyszin = helyszin;
        Datum = datum;
    }
}

Egyszerű magyarázat:
- public class Sportolo: létrehozunk egy saját típust
- get; set;: lekérdezhető és módosítható adattag
- konstruktor: amikor new Sportolo(...) történik, feltölti az adatokat

6. WINFORMS OPENFILEDIALOG + FÁJLBEOLVASÁS SABLON
=================================================

Form1.cs tetejére:

using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;
using System.Windows.Forms;

Form1 osztályba:

List<Fuggohid> hidak = new List<Fuggohid>();

Megnyitás menüpont eseményébe:

private void megnyitasToolStripMenuItem_Click(object sender, EventArgs e)
{
    OpenFileDialog ofd = new OpenFileDialog();
    ofd.Filter = "CSV fájlok|*.csv|Minden fájl|*.*";

    if (ofd.ShowDialog() == DialogResult.OK)
    {
        hidak.Clear();
        listBoxHidak.Items.Clear();

        StreamReader sr = new StreamReader(ofd.FileName, Encoding.UTF8);

        sr.ReadLine();

        while (!sr.EndOfStream)
        {
            string sor = sr.ReadLine();
            string[] adatok = sor.Split('\t');

            Fuggohid hid = new Fuggohid(
                int.Parse(adatok[0]),
                adatok[1],
                adatok[2],
                adatok[3],
                int.Parse(adatok[4]),
                int.Parse(adatok[5])
            );

            hidak.Add(hid);
            listBoxHidak.Items.Add(hid.Nev);
        }

        sr.Close();
    }
}

private void Form2_Load(object sender, EventArgs e)
{
    comboBoxOrszag.Items.Clear();

    var orszagok = hidak
        .Select(x => x.Orszag)
        .Distinct()
        .OrderBy(x => x);

    foreach (var orszag in orszagok)
    {
        comboBoxOrszag.Items.Add(orszag);
    }
}

private void buttonKereses_Click(object sender, EventArgs e)

14. PHP API ALAP SABLON

<?php

header("Content-Type: application/json; charset=utf-8");
header("Access-Control-Allow-Origin: *");

$kapcsolat = mysqli_connect("localhost", "root", "", "nevnapok");

if (!$kapcsolat) {
    echo json_encode(["hiba" => "adatbázis hiba"], JSON_UNESCAPED_UNICODE);
    exit;
}

mysqli_set_charset($kapcsolat, "utf8");

if (isset($_GET["nev"])) {

    $nev = mysqli_real_escape_string($kapcsolat, $_GET["nev"]);

    $sql = "SELECT * FROM nevnap WHERE nev1 = '$nev' OR nev2 = '$nev' LIMIT 1";

    $eredmeny = mysqli_query($kapcsolat, $sql);

    if (mysqli_num_rows($eredmeny) > 0) {
        $sor = mysqli_fetch_assoc($eredmeny);

        echo json_encode([
            "datum" => $sor["ho"] . "-" . $sor["nap"],
            "nevnap1" => $sor["nev1"],
            "nevnap2" => $sor["nev2"]
        ], JSON_UNESCAPED_UNICODE);
    } else {
        echo json_encode(["hiba" => "nincs találat"], JSON_UNESCAPED_UNICODE);
    }

    exit;
}

if (isset($_GET["nap"])) {

    $datum = $_GET["nap"];
    $darabok = explode("-", $datum);

    if (count($darabok) != 2) {
        echo json_encode(["hiba" => "nincs találat"], JSON_UNESCAPED_UNICODE);
        exit;
    }

    $ho = intval($darabok[0]);
    $nap = intval($darabok[1]);

    $sql = "SELECT * FROM nevnap WHERE ho = $ho AND nap = $nap";

    $eredmeny = mysqli_query($kapcsolat, $sql);

    if (mysqli_num_rows($eredmeny) > 0) {
        $sor = mysqli_fetch_assoc($eredmeny);

        echo json_encode([
            "datum" => $sor["ho"] . "-" . $sor["nap"],
            "nevnap1" => $sor["nev1"],
            "nevnap2" => $sor["nev2"]
        ], JSON_UNESCAPED_UNICODE);
    } else {
        echo json_encode(["hiba" => "nincs találat"], JSON_UNESCAPED_UNICODE);
    }

    exit;
}

echo json_encode([

15. JAVASCRIPT API HÍVÁS SABLON
===============================

Dátum alapján:

function keresesNapra() {
    let nap = document.getElementById("nap").value;
    let eredmeny = document.getElementById("eredmeny");

    if (nap == "") {
        eredmeny.innerHTML = "Adj meg egy dátumot!";
        return;
    }

    fetch("http://localhost/nevnapok/api/?nap=" + nap)
        .then(valasz => valasz.json())
        .then(adat => {
            if (adat.hiba) {
                eredmeny.innerHTML = "Nincs találat.";
            } else {
                eredmeny.innerHTML = "Névnap: " + adat.nevnap1 + " " + adat.nevnap2;
            }
        });
}

Név alapján:

function keresesNevre() {
    let nev = document.getElementById("nev").value;
    let eredmeny = document.getElementById("eredmeny");

    if (nev == "") {
        eredmeny.innerHTML = "Adj meg egy nevet!";
        return;
    }

    fetch("http://localhost/nevnapok/api/?nev=" + nev)
        .then(valasz => valasz.json())
        .then(adat => {
            if (adat.hiba) {
                eredmeny.innerHTML = "Nincs találat.";
            } else {
                eredmeny.innerHTML = nev + " névnapja: " + adat.datum;
            }
        });
}
